Hello World in Three Languages

main.cpp
#include <iostream>
int main(){
    std::cout << "Hello World!" << std::endl;
    return 0;
}
hello.py
def main():
    print("Hello World!")

if __name__ == "__main__":
    main()
Main.java
public class Main {
    public static void main(String[] args) {
        System.out.println("Hello World!");
    }
}
